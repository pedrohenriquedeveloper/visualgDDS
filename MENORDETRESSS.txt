Algoritmo "menordetresss"
Var
n1, n2, n3: inteiro

Inicio
escreval("Digite um número: ")
leia(n1)

escreval("Digite outro número: ")
leia(n2)

escreval("Digite um terceiro número: ")
leia(n3)

se (n1<n2) e (n1<n3) entao
   escreval("o menor número foi: ", n1)
senao
     se (n2<n1) e (n2<n3) entao
        escreval("o menor número foi: ", n2)
     senao
          se (n3<n1) e (n3<n2) entao
             escreval("o menor número foi: ", n3)
          senao
               se (n1=n2) e (n1=n3) entao
                  escreval("empate, o número repitido foi: ", n1)
               senao
                    se (n2=n1) e (n2=n3) entao
                       escreval("empate, o número repitido foi: ", n2)
                    senao
                         se (n3=n1) e (n3=n2) entao
                            escreval("empate, o número repitido foi: ", n3)
                         fimse
                    fimse
               fimse
          fimse
     fimse
fimse
