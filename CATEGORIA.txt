Algoritmo "Categoria"
Var
idade: inteiro


Inicio
escreval("Digite sua idade")
leia(idade)

se (idade >= 5) e (idade <= 7) entao
   escreval("Sua categoria é: Infantil 1")
senao
    se (idade >= 8) e (idade <= 10) entao
       escreval("Sua categoria é: Infantil 2")
    senao
         se (idade >= 11) e (idade <= 17) entao
            escreval("Sua categoria é: Juvenil")
         senao
              se idade >= 18 entao
                 escreval("Sua categoria é: adulto")
              fimse
         fimse
    fimse
fimse

Fimalgoritmo
